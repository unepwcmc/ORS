def admin_notification_questionnaire_submitted(user, questionnaire)
  recipients  User.administrators.map(&:email).join(",")
  from        "no-reply@unep-wcmc.org"
  subject     "[#{questionnaire.email_subject? ? questionnaire.email_subject : "Bern Convention Online Reporting System"}] Questionnaire submitted by #{user.full_name}"
  body        :user => user, :questionnaire => questionnaire
end
