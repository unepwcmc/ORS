def perform
  questionnaire = Questionnaire.find(questionnaire_id, :include => [{:questionnaire_parts => :part}, :questionnaire_fields])
  user = User.find(user_id)
  if user.role?(:admin) || questionnaire.status != 0
    SubmissionQuestionnaire.new.to_pdf questionnaire, user, self.url_prefix, self.short_version
    requester = User.find(requester_id)
    #mail the requester about the pdf being generated. (can be the user or an admin)
    UserMailer.deliver_pdf_generated(requester, questionnaire, user)
  end
end
