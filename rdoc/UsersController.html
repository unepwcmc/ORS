def create
  #@user = User.new(params[:user])
  #@questionnaire is fetch to send to the user .new page form. So that if the user is being added through an authorized submitters
  # page, the correct fields needed for that questionnaire are present in the form
  @questionnaire = Questionnaire.find(params[:questionnaire_id], :include => :questionnaire_fields) if params[:questionnaire_id]
  if @user.save
    @user.add_or_update_filtering_fields(params[:filtering_field]) if params[:filtering_field]
    if !current_user
      UserMailer.deliver_registration_confirmation(@user)
      UserMailer.deliver_admin_notification_registration_confirmation(@user)
    elsif @user.role?(:delegate)
      url = "http://#{request.host}/"
      I18n.locale = @user.language
      msg = "<p>#{t('generic.hi')} #{@user.full_name},</p><p>#{t('user_mailer.new_delegate.first_line')} #{t('title')}</p><p>#{t('user_mailer.new_delegate.access')} <a href='#{url}'>#{url}</a>.</p><p>#{t('user_mailer.new_delegate.login')}<br /><strong>#{t('generic.email')}:</strong> #{@user.email}<br /><strong>#{t('generic.password')}:</strong> #{params[:user][:password]}</p><p>#{t('user_mailer.greetings')}<br />#{t('title')}</p>"
      UserMailer.deliver_delegate_registration(@user, msg)#@user, params[:user][:password], "http://#{request.host}/")
    end
    respond_to do |format|
      format.html {
        if @user.creator_id == 0
          flash[:notice] = t('flash_messages.sign_up_success')
          redirect_to root_url
        else
          flash[:notice] = t('flash_messages.delegate_success')
          redirect_to user_delegates_path(current_user.id)
        end
      }
      format.js
    end
  else
    respond_to do |format|
      format.html { render :action => "new", :params => {:lang => (params[:lang]||"en")}}
      format.js
    end
  end
end

def delegate_section
  @user = User.find(params[:user_id])
  section = Section.find(params[:section_id])
  @authorization = current_user.authorization_for(section)
  raise CanCan::AccessDenied.new(t('flash_messages.not_authorized')) if !@authorization
  I18n.locale = @authorization[:language]
  @existing_delegations = section.delegations_from @user
  @available_delegates = Delegation.delegates_not_yet_delegated(@user, section)
  if !@available_delegates.empty?
    #flash[:error] = "Before delegating a section you need to define delegates in your home page. If you have done this, all of your delegates might already have been assigned with this section."
  #else
    @delegation_section = DelegationSection.new(:section_id => section.id)
  end
  respond_to do |format|
    format.js
  end
end

def destroy
  @user = User.find(params[:id])
  if @user.role?(:delegate)
    raise CanCan::AccessDenied.new(t('flash_messages.not_authorized')) if !current_user || (@user.creator != current_user)
    @delegator = @user.creator
    if @user.destroy
      flash[:notice] = "Delegate successfully destroyed."
    end
    respond_to do |format|
      format.html{ redirect_to user_delegates_path(@delegator)}
    end
  end
end

def group
  ( params[:group] && !params[:group].blank? ) ? @group = params[:group] : @group = params[:groups][:id]

    @users = User.add_users_to_group(params[:users], @group)
    @groups = User.group_counts
    @questionnaire = Questionnaire.find(params[:questionnaire_id]) if params[:questionnaire_id]
  end
  respond_to do |format|
    format.js
  end
end

def update
  #@user = User.find(params[:id])
  update_authorizations = params[:user][:language] != @user.language
  if @user.update_attributes(params[:user])
    @user.update_authorizations if update_authorizations
    @user.add_or_update_filtering_fields(params[:filtering_field]) if params[:filtering_field]
    flash[:notice] = t('flash_messages.update_profile_s') 
    redirect_to @user
  else
    render :action => 'edit'
  end
end

def upload_list
  @status = {}
  @user = current_user
  @user.parse_uploaded_list(params[:list_of_users].path, @status)
  if @status[:errors].present?
    flash[:error] = "There were some errors when parsing the file you provided."
  else
    flash[:notice] = "Successfully uploaded the list of users."
  end
end
