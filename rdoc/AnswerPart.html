def clone_answer_part(options_link, text_fields_link, numeric_answers_link)
  new_answer_part = self.clone_object([:answer_text])
  if self.field_type.is_a?(MultiAnswerOption)
    new_answer_part.field_type = options_link[self.field_type.id.to_s]
  elsif self.field_type.is_a?(TextAnswerField)
    new_answer_part.field_type = text_fields_link[self.field_type.id.to_s]
  elsif self.field_type.is_a?(NumericAnswer)
    new_answer_part.field_type = numeric_answers_link[self.field_type.id.to_s]
  end
  new_answer_part
end
