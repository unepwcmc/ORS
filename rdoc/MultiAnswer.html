def clone_answer_type options_link
  new_multi_answer = self.clone_object([:single, :other_required, :display_type])
  #answer_type_fields
  self.answer_type_fields.each do |field|
    new_multi_answer.answer_type_fields << field.clone_object([:language, :help_text, :is_default_language])
  end
  #clone the  other_fields if other_field is required
  if self.other_required?
    self.other_fields.each do |ofield|
      new_multi_answer.other_fields << ofield.clone_object([:language, :other_text, :is_default_language])
    end
  end
  #clone existing options
  self.multi_answer_options.each do |mao|
    new_multi_answer_option = MultiAnswerOption.new
    mao.multi_answer_option_fields.each do |field|
      new_multi_answer_option.multi_answer_option_fields << field.clone_object([:language, :is_default_language, :option_text])
    end
    new_multi_answer.multi_answer_options << new_multi_answer_option
    options_link[mao.id.to_s] = new_multi_answer_option #save the clone information, to keep the dependency on the copy of the questionnaire
  end
  new_multi_answer
end

def has_options_with_dependents?
  self.multi_answer_options.each do |mao|
    if mao.sections.present?
      return true #returns true when the multi answer option has dependent sections
    end
  end
  false
end
