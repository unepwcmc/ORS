def self.authorize_from_array_of_users(users_ids, questionnaire, url)
  users = []
  Array(users_ids).each do |user_id,| #users_ids is an hash with "user_id => user_id", coming from a set of check boxes
    user = User.find(user_id.to_i)
    if user
      authorized_submitter = AuthorizedSubmitter.new(:user_id => user.id, :questionnaire_id => questionnaire.id, :language => user.language)
      if authorized_submitter.save
        users += [user]
        UserMailer.deliver_authorisation_granted(user, questionnaire, url)
      end
    end
  end
  users
end

def self.remove_authorization_from_array_of_users(users_ids, questionnaire)
  users = []
  Array(users_ids).each do |user_id,| #users_ids is an hash with "user_id => user_id", coming from a set of checkboxes 
    user = User.find(user_id.to_i)
    if user
      authorized_submitter = questionnaire.authorized_submitters.find_by_user_id(user_id)
      if authorized_submitter
        users += [user]
        authorized_submitter.destroy
      end
    end
  end
  users
end
