def clone_answer_type text_fields_link
  new_text_answer = TextAnswer.new
  self.answer_type_fields.each do |field|
    new_text_answer.answer_type_fields << field.clone_object([:language, :help_text, :is_default_language])
  end
  self.text_answer_fields.each do |taf|
    new_taf = taf.clone_object([:width, :rows])
    new_text_answer.text_answer_fields << new_taf
    text_fields_link[taf.id.to_s] = new_taf
  end
  new_text_answer
end
