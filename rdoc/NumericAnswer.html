def clone_answer_type numeric_answers_link
  new_numeric_answer = self.clone_object([:max_value, :min_value])
  self.answer_type_fields.each do |field|
    new_numeric_answer.answer_type_fields << field.clone_object([:language, :help_text, :is_default_language])
  end
  numeric_answers_link[self.id.to_s] = new_numeric_answer
  new_numeric_answer
end
