def self.append_features(base)
  super

  self.class_eval do

    #check if the section/generated_section dependency is met (If the current_user's answers include the necessary answers for the
    #section to be visible!
    def dependency_condition_met?(user, loop_item_id=nil)
      #A Question can correspond to more than one generated question
      # so it is necessary to check if the condition is met for all the generated_questions' answers
      question = self.depends_on_question
      user_answers = user.answers.find_all_by_question_id_and_loop_item_id(question.id, loop_item_id)
      if not user_answers.present?
        return false
      end
      user_answers.each do |answer|
        #if not answer.loop_item.present? or question.section.available_for?(user, answer.loop_item)
        if self.depends_on_option_value?
          if !answer.answer_parts.find_by_field_type_id(self.depends_on_option_id)
            return false
          end
        else
          #if the option is present then the condition is not met, when the depends_on_option_value is false!
          if answer.answer_parts.find_by_field_type_id(self.depends_on_option_id)
            return false
          end
        end
        #end
      end
      return true
    end

    #=> Get the set of valid dependencies for a specific section
    # Get all the existing sections and remove section and its descendants
    def valid_dependencies
      questionnaire_root_sections = self.questionnaire.sections
      available_sections = []
      questionnaire_root_sections.each do |root|
        available_sections += root.self_and_descendants
      end
      #remove looping_sections that are not ancestors of self
      to_remove = []
      available_sections.each do |s|
        #remove branches from looping_sections that are not ancestors of self, to prevent a section from depending on multiple questions
        if s.looping? && !s.is_ancestor_of?(self)
          to_remove += s.self_and_descendants
          #else if the section s is looping and is ancestor of self
        elsif s.looping?
          #get the sections that exist between s and self
          gap_sections = s.descendants - self.descendants
          #and check that none of those is a looping section. If it is mark its ancestors to be removed
          #to prevent a section from depending on nested loopings.
          #TODO: It would be advised to create tests for this function to verify if it is working as it should.
          gap_sections.each do |gap|
            if gap.looping?
              to_remove += gap.ancestors
            end
          end
        end
      end
      available = available_sections - (self.self_and_descendants + to_remove)
      available.uniq.reject{|a| a.nil? }.sort{|a,b| a.questionnaire_part.lft <=> b.questionnaire_part.lft}
    end

    # return the section that/if self depends on, based on the depends_on_question value
    def section_that_this_depends_on
      self.depends_on_question ? self.depends_on_question.section.id : nil #""
    end

  end
