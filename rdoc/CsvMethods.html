def self.export the_section, submitters_ids, loop_sources, item=nil
  #puts "Dealing with section #{the_section.id} for item: #{item.try(:item_name)}"
  rows = []
  if !the_section.is_hidden? && !the_section.questions.any?
    next_row = Array.new(submitters_ids.size + 2)
    if item
      next_row[0] = "--"*the_section.level + OrtSanitize.white_space_cleanse(the_section.section_fields.find_by_is_default_language(true).loop_title(nil, item.loop_items.first))
    else
      next_row[0] = "--"*the_section.level + OrtSanitize.white_space_cleanse(the_section.title)
    end
    rows << next_row
  end
  the_section.questions.each do |q|
    rows += q.to_csv(submitters_ids, loop_sources, item)
  end
  the_section.children.each do |s|
    if s.looping? && !s.loop_item_type.filtering_field_id.present?
      item_names = s.loop_item_type.loop_item_names
      item_names.each do |item|
        loop_sources[s.loop_item_type.root.loop_source.id.to_s] = item
        rows += CsvMethods.export(s, submitters_ids, loop_sources, item)
      end
    else
      rows += CsvMethods.export s, submitters_ids, loop_sources, item
    end
  end
  rows
end

def self.fill_csv file_location, submitters, sections
  puts "Generating CSV file"
  puts "Opened file: #{file_location}"
  FasterCSV.open(file_location, "w") do |csv|
    #first row has the id's of the users that are authorized to fill a questionnaire.
    submitters_ids = submitters.sort{|a,b| a.first_name <=> b.first_name}.map(&:id) 
    submitters_head_line = submitters.sort{|a,b| a.first_name <=> b.first_name}.map{|a| "#{a.full_name} - #{a.email}"}
    csv << ["Section", "Question"] + submitters_head_line
    Array(sections).each do |section|
      loop_sources = {}
      if section.looping? && !section.loop_item_type.filtering_field_id.present?
        item_names = section.loop_item_type.loop_item_names
        item_names.each do |item|
          loop_sources[section.loop_item_type.root.loop_source.id.to_s] = item
          rows = CsvMethods.export(section, submitters_ids, loop_sources, item)
          rows.each do |row|
            csv << row
          end
        end
      else
        rows = CsvMethods.export(section, submitters_ids, loop_sources, nil)
        rows.each do |row|
          csv << row
        end
      end
    end
  end
  puts "CSV finished generating (#{file_location})"
  true
end
