def create
  @delegate = User.find(params[:delegate_id], :include => :creator)
  @delegation = Delegation.new(params[:delegation])
  if @delegation.save!
    I18n.locale = @delegate.language
    url = "http://#{request.host}/"
    msg = "<p>#{t('generic.hi')} #{@delegate.full_name },</p>
          <p>#{ t('user_mailer.questionnaire_delegated.intro') }</p>
          <p><strong>#{t('generic.questionnaire')}:</strong> #{@delegation.questionnaire.title }</p>
          <p>#{ t('user_mailer.questionnaire_delegated.access') }  <a href='#{url}/questionnaires/#{@delegation.questionnaire.id}/submission'>#{url}/questionnaires/#{ @delegation.questionnaire.id }/submission</a>.</p>
          <p>#{ t('user_mailer.questionnaire_delegated.dashboard') }</p>
          <p>#{ t('user_mailer.greetings')}<br />
          #{ t('title') }
          </p>"
    UserMailer.deliver_questionnaire_delegated(@delegate, @delegation, msg)
    flash[:notice] = "Task successfully delegated."
  else
    flash[:error] = "It was not possible to delegate the task."
  end
  respond_to do |format|
    format.html {redirect_to delegate_path(@delegate)}
  end
end

def destroy

  respond_to do |format|
    format.html { redirect_to delegate_path(@delegate)}

def new
  @delegate = User.find(params[:delegate_id], :include => {:creator => :available_questionnaires})
  @delegator = @delegate.creator
  @delegation = @delegator.delegations.build(:delegate_id => @delegate.id)
  @available_questionnaires = Delegation.available_questionnaires_for(@delegate, @delegator)
end
