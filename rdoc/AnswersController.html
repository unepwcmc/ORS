def add_document
  @question = Question.find(params[:id])
  @authorization = current_user.authorization_for(@question)
  raise CanCan::AccessDenied.new(t('flash_messages.not_authorized')) if !@authorization #|| @authorization[:status] == 2
  I18n.locale = @authorization[:language]
  @answer = Answer.find_or_create_by_question_id_and_questionnaire_id_and_user_id_and_loop_item_id(@question.id, @question.section.root.questionnaire.id, current_user.id, params[:loop_item])
  @answer.documents.build
  respond_to do |format|
    format.js
  end
end

def update
  @answer = Answer.find(params[:id])
  #raise an AccessDenied Exception if the answer doesn't belong to this user (if its a respondent)
  #or if it wasn't last edited by the current_user (so that the delegates can update their own answers
  #but not the respondets answers.)
  #TODO: Verify if this is desired behavior
  raise CanCan::AccessDenied.new(t('flash_messages.not_authorized')) if @answer.user != current_user && @answer.last_editor != current_user
  if params[:to_render] == "save_links"
    Answer.validate_the_urls(params[:answer])
  end
  if @answer.update_attributes(params[:answer])
    flash[:notice] = "Changes successfully made."
    @success = true
  else
    flash[:error] = "There was a problem when saving your changes, please try again."
    @success = false
  end
  respond_to do |format|
    format.js { render params[:to_render] }
  end
end
